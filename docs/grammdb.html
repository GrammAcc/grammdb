def select( columns: str | tuple[str, ...], from_clause: sqlalchemy.sql.selectable.Select, *where_clauses: WhereFunc) -> sqlalchemy.sql.selectable.Select: 
22def select(
23    columns: str | tuple[str, ...], from_clause: sa.Select, *where_clauses: WhereFunc
24) -> sa.Select:
25    """Wrapper for generating sqlalchemy select statements with a more sql-like DSL.
26
27    Note:
28        Using "*" as the first argument is a special case that does not actually do a select star
29        in SQL. It is simply syntax sugar around SQLAlchemy's regular behavior of selecting the
30        full set of columns in the from clause by name.
31        In other words, `sa.select(MyTable)` and `grammdb.select("*", sa.select(MyTable))` are
32        effectively equivalent.
33
34    Args:
35        columns:
36            One of the following:
37                A single string representing the name of the column to select.
38                A literal "*" meaning select all columns in the from clause.
39                A list of strings representing the column names to select from the from clause.
40        from_clause:
41            A selectable object. Usually, this would be provided using a from function to create
42            a statement that reads more like sql.
43        *where_clauses:
44            Arbitrary number of where functions to apply as the where clause in the built query.
45    """
46
47    # Explanation for Any:
48    # SQLAlchemy's core is too dynamic, and it's type annotations don't match the documented
49    # inputs. E.g. a tuple of string column names doesn't satisfy the type annotations for
50    # with_only_columns below, but this works at runtime.
51    select_columns: Any = (
52        from_clause.selected_columns if columns == "*" else from_clause.selected_columns[columns]
53    )
54    return functools.reduce(
55        _reducer,
56        where_clauses,
57        from_clause.with_only_columns(select_columns),
58    )
